import fitz

def extract_invoice_lines(pdf_path):
    doc = fitz.open(pdf_path)
    items = []

    for page in doc:
        blocks = page.get_text("blocks")

        for block in blocks:
            text = block[4]

            # détecter une ligne de tableau produit
            if text.strip().startswith(tuple(str(i) for i in range(1, 300))):
                parts = text.split("\n")

                # Exemple de ligne :
                # Pos. Hoeveelheid Brutoprijs Nettoprijs Netto waarde
                # 10 ...
                if len(parts) >= 2:
                    pos = parts[0].split()[0]
                    # détecter article id
                    artikel = None
                    for p in parts:
                        if p.startswith("Artikel:"):
                            artikel = p.replace("Artikel:", "").strip()

                    # quantité = deuxième ligne typiquement
                    qty_line = parts[1].split()

                    items.append({
                        "pos": pos,
                        "artikel": artikel,
                        "qty": qty_line[1],
                        "brutoprijs": qty_line[2],
                        "nettoprijs": qty_line[3],
                        "netto_waarde": qty_line[4]
                    })

    return items
📦 Extraction du bon de livraison
def extract_delivery_lines(pdf_path):
    doc = fitz.open(pdf_path)
    items = []

    for page in doc:
        blocks = page.get_text("blocks")

        for b in blocks:
            text = b[4].strip()

            # détecter les lignes produit
            if text.startswith(tuple(str(i) for i in range(1, 500))):
                parts = text.split()
                pos = parts[0]
                artikel = parts[1]
                ean = parts[2]
                qty = parts[-4]  # ex : "8 ST"
                qty_val = qty

                items.append({
                    "pos": pos,
                    "artikel": artikel,
                    "ean": ean,
                    "qty_livree": qty_val
                })

    return items
🔥 5. Comparaison automatique (le cœur de ton système)
Une fois les 2 listes extraites :

invoice = extract_invoice_lines("INV_IN_ATT_PID_30055.pdf")
delivery = extract_delivery_lines("9209140563.pdf")

def compare(invoice, delivery):
    discrepancies = []

    delivery_map = {item['artikel']: item for item in delivery}

    for inv in invoice:
        ref = inv["artikel"]

        if ref not in delivery_map:
            discrepancies.append({
                "artikel": ref,
                "issue": "Facturé mais pas livré"
            })
            continue

        q_inv = float(inv["qty"])
        q_del = float(delivery_map[ref]["qty_livree"])

        if q_inv != q_del:
            discrepancies.append({
                "artikel": ref,
                "issue": f"Quantité différente: facture={q_inv}, livraison={q_del}"
            })

    return discrepancies

# utils/extractor.py
import fitz
import io
from typing import Dict, Any

def extract_pdf_raw(path: str) -> Dict[str, Any]:
    doc = fitz.open(path)
    pages = []
    full_text = []

    for i, page in enumerate(doc):
        text = page.get_text()
        blocks = page.get_text("blocks")
        words = page.get_text("words")

        pages.append({
            "page_number": i+1,
            "text": text,
            "blocks": blocks,
            "words": words,
        })
        full_text.append(text)

    return {
        "page_count": len(doc),
        "pages": pages,
        "full_text": "\n".join(full_text)
    }

def is_scanned(pdf_raw: Dict[str, Any]) -> bool:
    # heuristique : si la plupart des pages ont peu ou pas de texte
    texts = [p['text'].strip() for p in pdf_raw['pages']]
    empty = sum(1 for t in texts if len(t) < 50)
    return empty / max(1, len(texts)) > 0.5
parsers/generic.py
Parser générique qui tente d'extraire lignes de type ARTICLE/QTY/PRICE/EAN.

# parsers/generic.py
EAN_REGEX = re.compile(r"\b\d{13}\b")
PRICE_REGEX = re.compile(r"\d+[\.,]\d{2}")

def normalize_number(s: str) -> str:
    return s.replace(',', '.').replace('\u00A0', '').strip()


def find_ean(text: str):
    m = EAN_REGEX.search(text)
    return m.group(0) if m else None


def simple_line_parser(page_text: str) -> List[Dict]:
    """Cherche des lignes contenant EAN ou pattern 'SKU ... qty ... price'"""
    items = []
    lines = [l for l in page_text.splitlines() if l.strip()]

    for line in lines:
        ean = find_ean(line)
        # heuristique : si ligne commence par un index numérique
        if re.match(r"^\s*\d+\s", line) or ean:
            # extraire nombres (prix/qty) de la ligne
            prices = PRICE_REGEX.findall(line)
            numbers = re.findall(r"\d+[\.,]?\d*", line)

            # tentative d'extraction
            item = {
                "raw": line,
                "ean": ean,
                "sku": None,
                "description": None,
                "qty": None,
                "unit_price": None,
                "total": None
            }

            parts = line.split()
            # sku candidate = deuxième token si la ligne commence par pos
            if re.match(r"^\s*\d+\s", line) and len(parts) > 1:
                item['sku'] = parts[1]

            # qty heuristique : look for patterns like '8 ST' or '8 PC' or last integer before 'KG'
            qty_search = re.search(r"(\d+[\,\.]?\d*)\s*(ST|PC|KG|PAC|PAX)?", line)
            if qty_search:
                item['qty'] = normalize_number(qty_search.group(1))

            # price heuristique : last price-looking token
            if prices:
                item['unit_price'] = normalize_number(prices[-1])

            items.append(item)

    return items


def parse(pdf_raw: Dict) -> Dict:
    """Retourne dict: {items: [...], metadata: {...}}"""
    items = []
    for p in pdf_raw['pages']:
        items += simple_line_parser(p['text'])

    return {"items": items, "metadata": {"page_count": pdf_raw['page_count']}}

# parsers/knauf.py
import re
from typing import Dict, List
from utils.extractor import extract_pdf_raw

PRICE_RE = re.compile(r"(\d+[\.,]\d{2})\s*/(PAC|PC|KG)?")


def parse(pdf_raw: Dict) -> Dict:
    items = []

    for p in pdf_raw['pages']:
        lines = [l for l in p['text'].splitlines() if l.strip()]

        # On parcourt les lignes et on cherche celles qui commencent par un numéro de position
        for i, line in enumerate(lines):
            if re.match(r"^\s*\d+\s", line):
                # concatérer quelques lignes suivantes (parfois la description est multiline)
                block = line
                for j in range(1,3):
                    if i+j < len(lines) and not re.match(r"^\s*\d+\s", lines[i+j]):
                        block += ' ' + lines[i+j]

                # extraire l'article
                m_art = re.search(r"Artikel:\s*(\w+)", block)
                sku = m_art.group(1) if m_art else None

                # ean
                m_ean = re.search(r"(\d{13})", block)
                ean = m_ean.group(1) if m_ean else None

                # qty: look for pattern '10 PAC' or '8 PC' or '4 PC 20 KG'
                m_qty = re.search(r"(\d+)\s*(PAC|PC|ST|KG)?", block)
                qty = int(m_qty.group(1)) if m_qty else None

                # price: look for format '3,17 /PAC 3,17 /PAC 31,70' -> We take the nettoprijs
                m_prices = re.findall(r"(\d+[\.,]\d{2})", block)
                unit_price = None
                total = None
                if m_prices:
                    # heuristic: last is total, second-last is unit price
                    if len(m_prices) >= 2:
                        unit_price = m_prices[-2]
                        total = m_prices[-1]
                    else:
                        unit_price = m_prices[-1]

                items.append({
                    "sku": sku,
                    "ean": ean,
                    "description": block,
                    "qty": qty,
                    "unit_price": unit_price,
                    "line_total": total
                })

    return {"items": items}

# api/main.py
import os
from fastapi import FastAPI, UploadFile, File, HTTPException
from fastapi.responses import JSONResponse
from utils.extractor import extract_pdf_raw, is_scanned
from importlib import import_module
from pathlib import Path

app = FastAPI(title='AutoInvoiceParser')

PARSERS = {
    'knauf': 'parsers.knauf',
    'generic': 'parsers.generic'
}


def detect_fournisseur(text: str):
    # heuristique simple (à enrichir)
    if 'Knauf' in text or 'Rue du Parc Industriel' in text:
        return 'knauf'
    return 'generic'


@app.post('/parse')
async def parse_pdf(file: UploadFile = File(...)):
    if not file.filename.lower().endswith('.pdf'):
        raise HTTPException(status_code=400, detail='Only PDF supported')

    tmp = f"/tmp/{file.filename}"
    content = await file.read()
    with open(tmp, 'wb') as f:
        f.write(content)

    pdf_raw = extract_pdf_raw(tmp)
    text = pdf_raw['full_text']
    fournisseur = detect_fournisseur(text)

    module_path = PARSERS.get(fournisseur, 'parsers.generic')
    mod = import_module(module_path)
    result = mod.parse(pdf_raw)

    # suppression du tmp
    try:
        os.remove(tmp)
    except Exception:
        pass

    return JSONResponse(result)

✅ 1. Code C# (.NET Core) — Service pour consommer ton microservice Python (FastAPI)
Ce service C# utilise HttpClient, envoie un PDF à /parse, et récupère les items normalisés.

📁 Services/InvoiceParserService.cs

using System.Net.Http;
using System.Net.Http.Headers;
using System.Threading.Tasks;
using Newtonsoft.Json;
using System.Collections.Generic;

public class ParsedItem
{
    public string Sku { get; set; }
    public string Ean { get; set; }
    public string Description { get; set; }
    public double? Qty { get; set; }
    public double? Unit_Price { get; set; }
    public double? Line_Total { get; set; }
}

public class ParseResult
{
    public List<ParsedItem> Items { get; set; }
}

public interface IInvoiceParserService
{
    Task<ParseResult> ParsePdfAsync(byte[] pdfData, string fileName);
}

public class InvoiceParserService : IInvoiceParserService
{
    private readonly HttpClient _httpClient;

    public InvoiceParserService(HttpClient httpClient)
    {
        _httpClient = httpClient;
        _httpClient.BaseAddress = new Uri("http://localhost:8000");
    }

    public async Task<ParseResult> ParsePdfAsync(byte[] pdfData, string fileName)
    {
        using var form = new MultipartFormDataContent();
        var fileContent = new ByteArrayContent(pdfData);
        fileContent.Headers.ContentType = MediaTypeHeaderValue.Parse("application/pdf");

        form.Add(fileContent, "file", fileName);

        var response = await _httpClient.PostAsync("/parse", form);
        response.EnsureSuccessStatusCode();

        var json = await response.Content.ReadAsStringAsync();
        return JsonConvert.DeserializeObject<ParseResult>(json);
    }
}

📁 src/app/services/pdf-parser.service.ts

import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface ParsedItem {
  sku: string;
  ean: string;
  description: string;
  qty: number;
  unit_price: number;
  line_total: number;
}

export interface ParseResult {
  items: ParsedItem[];
}

@Injectable({
  providedIn: 'root'
})
export class PdfParserService {

  private apiUrl = 'http://localhost:8000/parse';

  constructor(private http: HttpClient) {}

  parsePdf(file: File): Observable<ParseResult> {
    const formData = new FormData();
    formData.append('file', file, file.name);
    return this.http.post<ParseResult>(this.apiUrl, formData);
  }
}

import json

def compare(invoice_items, delivery_items):
    inv_map = {item["sku"]: item for item in invoice_items if item["sku"]}
    del_map = {item["sku"]: item for item in delivery_items if item["sku"]}

    results = {
        "missing_in_delivery": [],
        "missing_in_invoice": [],
        "qty_mismatch": [],
        "ok": []
    }

    # Articles facturés mais pas livrés
    for sku, inv in inv_map.items():
        if sku not in del_map:
            results["missing_in_delivery"].append({
                "sku": sku,
                "invoice_qty": inv.get("qty")
            })
        else:
            # Vérif quantités
            inv_qty = float(inv.get("qty", 0))
            del_qty = float(del_map[sku].get("qty", 0))
            if inv_qty != del_qty:
                results["qty_mismatch"].append({
                    "sku": sku,
                    "invoice_qty": inv_qty,
                    "delivery_qty": del_qty
                })
            else:
                results["ok"].append({
                    "sku": sku,
                    "qty": inv_qty
                })

    # Articles livrés mais pas facturés
    for sku, del_item in del_map.items():
        if sku not in inv_map:
            results["missing_in_invoice"].append({
                "sku": sku,
                "delivery_qty": del_item.get("qty")
            })

    return results


if __name__ == "__main__":
    # Exemple d'usage
    invoice_json = json.load(open("invoice.json"))
    delivery_json = json.load(open("delivery.json"))

    result = compare(invoice_json["items"], delivery_json["items"])
    print(json.dumps(result, indent=4, ensure_ascii=False))
