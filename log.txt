class ValueValidator:
    """Valide les valeurs extraites pour éviter les faux positifs"""
    
    @staticmethod
    def is_valid_invoice_number(text: Optional[str]) -> bool:
        if not text or len(text) < 2:
            return False
        text_lower = text.lower()
        # Rejeter si contient des mots de TVA
        invalid_patterns = ["tva", "vat", "btw", "%", "intracom", "taux", "rate"]
        if any(p in text_lower for p in invalid_patterns):
            return False
        # Doit contenir au moins un chiffre ou alphanumérique
        has_digit = any(c.isdigit() for c in text)
        return has_digit
    
    @staticmethod
    def is_valid_amount(text: Optional[str], context: str = "unknown") -> bool:
        """Valide qu'un montant est cohérent selon le contexte"""
        if not text:
            return False
        
        # Nettoyage pour analyse
        cleaned = text.replace(",", ".").replace(" ", "").replace("€", "").replace("$", "")
        
        try:
            value = float(cleaned)
        except:
            return False
        
        # Validation contextuelle
        if context == "unit_price":
            # Prix unitaire: généralement entre 0.01 et 100000
            return 0.01 <= value <= 100000
        elif context == "total_line":
            # Total ligne: doit être cohérent avec quantité * PU
            return 0.01 <= value <= 1000000
        elif context == "global_total":
            # Total global: généralement le plus grand montant
            return 10 <= value <= 10000000
        elif context == "quantity":
            # Quantité: entier positif raisonnable
            return 1 <= value <= 100000 and value == int(value)
        
        return value > 0
    
    @staticmethod
    def normalize_amount(text: str, context: str = "unknown") -> Optional[float]:
        """Extrait et normalise un montant avec validation"""
        if not text:
            return None
        
        # Stratégie: trouver tous les nombres et sélectionner selon contexte
        import re  # Uniquement pour extraction de nombres, pas de pattern matching complexe
        
        # Patterns de séparation décimale par langue
        # FR/DE: 1.234,56 ou 1234,56
        # US/UK: 1,234.56 ou 1234.56
        
        candidates = []
        
        # Extraction de tous les nombres potentiels
        # Format avec séparateur de milliers puis décimal: 1.234,56 ou 1,234.56
        parts = text.replace("€", "").replace("$", "").split()
        
        for part in parts:
            # Nettoyer le segment
            clean = part.strip().replace("(", "").replace(")", "")
            
            # Détection du format
            if "," in clean and "." in clean:
                # Format mixte: déterminer lequel est le décimal
                last_comma = clean.rfind(",")
                last_dot = clean.rfind(".")
                
                if last_comma > last_dot:
                    # Format européen: 1.234,56
                    clean = clean.replace(".", "").replace(",", ".")
                else:
                    # Format anglo-saxon: 1,234.56
                    clean = clean.replace(",", "")
            elif "," in clean:
                # Peut être décimal (FR) ou milliers (US)
                # Si 3 chiffres après la dernière virgule -> décimal
                after_comma = clean.split(",")[-1]
                if len(after_comma) == 2:
                    clean = clean.replace(",", ".")
                else:
                    clean = clean.replace(",", "")
            
            try:
                val = float(clean)
                candidates.append(val)
            except:
                continue
        
        if not candidates:
            return None
        
        # Sélection selon contexte
        if context == "quantity":
            # Prendre le plus petit entier
            ints = [c for c in candidates if c == int(c) and 1 <= c <= 100000]
            return min(ints) if ints else None
        
        elif context == "unit_price":
            # Prix unitaire: généralement entre 0.1 et 50000, pas le plus grand
            valid = [c for c in candidates if 0.01 <= c <= 50000]
            # Si plusieurs, prendre celui qui ressemble le plus à un prix unitaire
            # (pas un grand nombre qui serait un total)
            if valid:
                # Trier et prendre un des premiers si plusieurs candidats
                valid.sort()
                return valid[0] if len(valid) == 1 else valid[1] if len(valid) > 1 else valid[0]
            return None
        
        elif context in ["total_line", "global_total"]:
            # Prendre le plus grand montant valide
            valid = [c for c in candidates if 0.01 <= c <= 10000000]
            return max(valid) if valid else None
        
        return candidates[0] if candidates else None

class RobustPDFExtractor:
    def __init__(self, pdf_path: str):
        self.doc = fitz.open(pdf_path)
        self.page = self.doc[0]
        self.words = self._get_words_with_positions()
        self.lines = self._group_by_lines()
        self.validator = ValueValidator()
        
    def _get_words_with_positions(self) -> List[Dict]:
        words = []
        for block in self.page.get_text("words"):
            x0, y0, x1, y1, word, block_no, line_no, word_no = block
            words.append({
                "text": word,
                "text_lower": word.lower(),
                "bbox": (x0, y0, x1, y1),
                "x_center": (x0 + x1) / 2,
                "y_center": (y0 + y1) / 2,
                "line_no": line_no,
                "block_no": block_no
            })
        return words
    
    def _group_by_lines(self) -> Dict[int, List[Dict]]:
        """Regroupe les mots par ligne avec tolérance"""
        lines = defaultdict(list)
        for word in self.words:
            # Clé de ligne basée sur Y avec tolérance de 5 points
            line_key = round(word["y_center"] / 5) * 5
            lines[line_key].append(word)
        
        # Trier chaque ligne par X
        for line_key in lines:
            lines[line_key].sort(key=lambda w: w["x_center"])
        
        return dict(lines)
    
    def _find_field_with_context(self, field_config: Dict) -> Optional[str]:
        """
        Extraction avec validation contextuelle stricte
        """
        labels = field_config["labels"]
        direction = field_config.get("direction", "right")
        max_dist = field_config.get("max_distance", 100)
        pos_context = field_config.get("context_positive", [])
        neg_context = field_config.get("context_negative", [])
        exclude_patterns = field_config.get("exclude_if_contains", [])
        
        candidates = []
        
        for word in self.words:
            # Vérifier si le mot correspond à un label
            if not any(label in word["text_lower"] for label in labels):
                continue
            
            # Vérifier contexte négatif immédiat (rejet rapide)
            line_words = self._get_words_on_same_line(word)
            line_text = " ".join([w["text_lower"] for w in line_words])
            
            if any(neg in line_text for neg in neg_context):
                continue
            
            # Chercher la valeur selon la direction
            value = self._extract_value_directional(word, direction, max_dist)
            
            if not value:
                continue
            
            # Vérifier patterns d'exclusion dans la valeur
            if any(pat in value.lower() for pat in exclude_patterns):
                continue
            
            # Vérifier contexte positif (score de confiance)
            confidence = 0
            nearby_text = self._get_nearby_text(word, radius=150)
            
            if any(pos in nearby_text for pos in pos_context):
                confidence += 2
            
            # Validation spécifique au champ
            if field_config.get("validate"):
                if not field_config["validate"](value):
                    continue
                confidence += 1
            
            candidates.append({
                "value": value,
                "confidence": confidence,
                "source_word": word
            })
        
        if not candidates:
            return None
        
        # Trier par confiance décroissante
        candidates.sort(key=lambda x: x["confidence"], reverse=True)
        return candidates[0]["value"]
    
    def _get_words_on_same_line(self, reference: Dict, tolerance: int = 5) -> List[Dict]:
        """Récupère tous les mots sur la même ligne"""
        ref_y = reference["y_center"]
        return [w for w in self.words if abs(w["y_center"] - ref_y) < tolerance]
    
    def _extract_value_directional(self, ref: Dict, direction: str, max_dist: float) -> Optional[str]:
        """Extrait la valeur dans une direction donnée"""
        candidates = []
        
        for word in self.words:
            if word == ref:
                continue
            
            dx = word["x_center"] - ref["x_center"]
            dy = word["y_center"] - ref["y_center"]
            dist = (dx**2 + dy**2)**0.5
            
            if dist > max_dist:
                continue
            
            # Vérifier direction
            valid = False
            if direction == "right" and dx > 0 and abs(dy) < 30:
                valid = True
            elif direction == "below" and dy > 0 and abs(dx) < 100:
                valid = True
            elif direction == "left" and dx < 0 and abs(dy) < 30:
                valid = True
            elif direction == "above" and dy < 0 and abs(dx) < 100:
                valid = True
            
            if valid:
                candidates.append((word, dist))
        
        if not candidates:
            return None
        
        # Trier par distance et concaténer les mots proches
        candidates.sort(key=lambda x: x[1])
        
        # Prendre les mots dans une fenêtre de proximité
        result_words = [candidates[0][0]]
        base_x = candidates[0][0]["x_center"]
        
        for word, dist in candidates[1:3]:  # Limite à 3 mots supplémentaires
            if abs(word["x_center"] - base_x) < 50:  # Proche horizontalement
                result_words.append(word)
                base_x = word["x_center"]
        
        result_words.sort(key=lambda w: w["x_center"])
        return " ".join([w["text"] for w in result_words])
    
    def _get_nearby_text(self, ref: Dict, radius: float = 100) -> str:
        """Récupère le texte dans un rayon donné"""
        nearby = []
        for word in self.words:
            dist = ((word["x_center"] - ref["x_center"])**2 + 
                   (word["y_center"] - ref["y_center"])**2)**0.5
            if dist < radius:
                nearby.append(word["text_lower"])
        return " ".join(nearby)
    
    def detect_document_type(self) -> DocumentType:
        """Détection avec scoring pondéré"""
        full_text = " ".join([w["text_lower"] for w in self.words])
        
        scores = defaultdict(int)
        
        for doc_type, keywords in KEYWORDS["typeDocument"].items():
            for kw in keywords:
                if kw in full_text:
                    # Poids plus fort pour correspondance exacte
                    if f" {kw} " in f" {full_text} ":
                        scores[doc_type] += 2
                    else:
                        scores[doc_type] += 1
        
        if not scores or max(scores.values()) == 0:
            return DocumentType.UNKNOWN
        
        best = max(scores, key=scores.get)
        return best if scores[best] >= 2 else DocumentType.UNKNOWN
    
    def extract_invoice_number(self) -> Optional[str]:
        """Extraction robuste du numéro de facture"""
        config = KEYWORDS["fields"]["numero"].copy()
        config["validate"] = self.validator.is_valid_invoice_number
        
        # Tentative 1: Par contexte
        result = self._find_field_with_context(config)
        if result:
            return self._clean_invoice_number(result)
        
        # Tentative 2: Zone top-right (où se trouve généralement le numéro)
        zone_text = self._extract_from_zone("top_right")
        if zone_text:
            # Chercher un pattern numérique valide dans la zone
            words = zone_text.split()
            for word in words:
                if self.validator.is_valid_invoice_number(word):
                    return self._clean_invoice_number(word)
        
        return None
    
    def _clean_invoice_number(self, text: str) -> Optional[str]:
        """Nettoie le numéro de facture"""
        if not text:
            return None
        
        # Garder alphanumériques et tirets
        cleaned = "".join(c for c in text if c.isalnum() or c in "-_/").strip("-_/")
        
        # Doit contenir au moins un chiffre
        if not any(c.isdigit() for c in cleaned):
            return None
        
        return cleaned if len(cleaned) >= 2 else None
    
    def extract_date(self) -> Optional[str]:
        """Extraction de date avec résolution d'ambiguïté"""
        config = KEYWORDS["fields"]["date"]
        
        # Tentative avec contexte
        date_str = self._find_field_with_context(config)
        if date_str:
            return self._normalize_date(date_str)
        
        # Recherche de pattern date dans tout le document (top half)
        top_words = [w for w in self.words if w["y_center"] < self.page.rect.height * 0.4]
        date_candidates = self._find_date_patterns(top_words)
        
        if date_candidates:
            # Prendre la date la plus proche du haut (généralement date de facture)
            return date_candidates[0]
        
        return None
    
    def _find_date_patterns(self, words: List[Dict]) -> List[str]:
        """Trouve tous les patterns de date et les normalise"""
        # Concaténer pour analyse
        text = " ".join([w["text"] for w in words])
        
        dates = []
        parts = text.replace("-", "/").replace(".", "/").split()
        
        for part in parts:
            if "/" in part or "-" in part:
                normalized = self._normalize_date(part)
                if normalized and normalized != part:
                    dates.append(normalized)
        
        return dates
    
    def _normalize_date(self, date_str: str) -> Optional[str]:
        """
        Normalise une date avec heuristiques pour résoudre MM/DD vs DD/MM
        """
        if not date_str:
            return None
        
        # Nettoyage
        d = date_str.strip().replace("-", "/").replace(".", "/").replace(" ", "/")
        
        parts = d.split("/")
        if len(parts) != 3:
            return None
        
        # Essayer d'interpréter
        p1, p2, p3 = parts[0], parts[1], parts[2]
        
        # Déterminer l'année (généralement la dernière ou première, 4 chiffres)
        year = None
        year_pos = None
        for i, p in enumerate([p1, p2, p3]):
            if len(p) == 4 and p.isdigit() and 1900 < int(p) < 2100:
                year = p
                year_pos = i
                break
        
        if not year:
            # Année sur 2 chiffres
            for i, p in enumerate([p1, p2, p3]):
                if len(p) == 2 and p.isdigit():
                    year = "20" + p if int(p) < 50 else "19" + p
                    year_pos = i
                    break
        
        if not year:
            return None
        
        # Déterminer jour et mois
        remaining = [(p1, 0), (p2, 1), (p3, 2)]
        remaining = [r for r in remaining if r[1] != year_pos]
        
        if len(remaining) != 2:
            return None
        
        (val_a, pos_a), (val_b, pos_b) = remaining
        
        if not (val_a.isdigit() and val_b.isdigit()):
            return None
        
        a, b = int(val_a), int(val_b)
        
        # Heuristiques de résolution
        day, month = None, None
        
        # Si un > 12, c'est forcément le jour
        if a > 12:
            day, month = a, b
        elif b > 12:
            day, month = b, a
        elif a > 12 and b > 12:
            # Impossible, date invalide
            return None
        else:
            # Ambiguïté: utiliser le contexte du fournisseur/langue si disponible
            # Par défaut: format européen DD/MM/YYYY
            day, month = a, b
            
            # Si on détecte des indices US (ex: "Invoice Date" plutôt que "Date")
            us_indicators = ["invoice date", "mm/dd", "due date"]
            full_text = " ".join([w["text_lower"] for w in self.words[:50]])
            if any(ind in full_text for ind in us_indicators):
                day, month = b, a
        
        return f"{str(day).zfill(2)}/{str(month).zfill(2)}/{year}"
    
    def extract_products(self) -> List[Dict]:
        """Extraction robuste des lignes de produits"""
        # Détecter la zone du tableau
        table_zone = self._detect_table_zone()
        if not table_zone:
            return []
        
        # Extraire les lignes dans cette zone
        lines_in_table = []
        y_min, y_max = table_zone["y_range"]
        
        for y_key, line_words in self.lines.items():
            if y_min <= y_key <= y_max:
                lines_in_table.append((y_key, line_words))
        
        # Trier par position Y
        lines_in_table.sort(key=lambda x: x[0])
        
        # Identifier la ligne d'en-tête
        header_line_idx = self._find_header_line(lines_in_table)
        if header_line_idx is None:
            return []
        
        # Analyser la structure des colonnes à partir de l'en-tête
        column_structure = self._analyze_column_structure(lines_in_table[header_line_idx][1])
        
        products = []
        for y_key, line_words in lines_in_table[header_line_idx + 1:]:
            product = self._parse_product_line_structured(line_words, column_structure)
            if product and product.get("description"):
                products.append(product)
            
            # Arrêt si on atteint les totaux
            line_text = " ".join([w["text_lower"] for w in line_words])
            if any(stop in line_text for stop in KEYWORDS["fields"]["montant_total"]["labels"]):
                break
        
        return products
    
    def _detect_table_zone(self) -> Optional[Dict]:
        """Détecte la zone probable du tableau de produits"""
        # Chercher les mots-clés de tableau
        table_indicators = (KEYWORDS["table_headers"]["quantity"]["labels"] + 
                          KEYWORDS["table_headers"]["description"]["labels"])
        
        y_positions = []
        for word in self.words:
            if any(ind in word["text_lower"] for ind in table_indicators):
                y_positions.append(word["y_center"])
        
        if not y_positions:
            return None
        
        # La table commence généralement après l'en-tête
        table_start = min(y_positions) - 20
        table_end = self.page.rect.height * 0.8  # Fin approximative
        
        return {"y_range": (table_start, table_end)}
    
    def _find_header_line(self, lines: List[Tuple[int, List[Dict]]]) -> Optional[int]:
        """Trouve la ligne d'en-tête du tableau"""
        for idx, (y_key, words) in enumerate(lines):
            line_text = " ".join([w["text_lower"] for w in words])
            
            # Chercher plusieurs indicateurs de colonnes
            indicators = 0
            for header_type, config in KEYWORDS["table_headers"].items():
                if any(label in line_text for label in config["labels"]):
                    indicators += 1
            
            if indicators >= 2:  # Au moins 2 colonnes identifiées
                return idx
        
        return None
    
    def _analyze_column_structure(self, header_words: List[Dict]) -> Dict[str, Tuple[float, float]]:
        """Analyse la position des colonnes à partir de l'en-tête"""
        columns = {}
        
        for header_type, config in KEYWORDS["table_headers"].items():
            for word in header_words:
                if any(label in word["text_lower"] for label in config["labels"]):
                    x_center = word["x_center"]
                    # Définir une plage pour cette colonne
                    columns[header_type] = (x_center - 50, x_center + 50)
                    break
        
        return columns
    
    def _parse_product_line_structured(self, words: List[Dict], column_structure: Dict) -> Dict:
        """Parse une ligne en utilisant la structure de colonnes connue"""
        # Assigner chaque mot à une colonne
        cols = defaultdict(list)
        
        for word in words:
            x = word["x_center"]
            assigned = False
            
            for col_name, (x_min, x_max) in column_structure.items():
                if x_min <= x <= x_max:
                    cols[col_name].append(word)
                    assigned = True
                    break
            
            if not assigned:
                # Assigner à la colonne la plus proche
                closest = min(column_structure.items(), 
                            key=lambda c: abs(x - (c[1][0] + c[1][1])/2))
                cols[closest[0]].append(word)
        
        # Extraire les valeurs avec validation contextuelle
        quantity_words = cols.get("quantity", [])
        desc_words = cols.get("description", [])
        price_words = cols.get("unit_price", [])
        total_words = cols.get("total_line", [])
        
        # Quantité: entier positif
        qty_text = " ".join([w["text"] for w in quantity_words])
        quantity = self.validator.normalize_amount(qty_text, "quantity") or 1
        
        # Description: texte libre
        description = " ".join([w["text"] for w in desc_words]).strip()
        
        # Prix unitaire: validation stricte
        price_text = " ".join([w["text"] for w in price_words])
        unit_price = self.validator.normalize_amount(price_text, "unit_price")
        
        # Total ligne: doit être cohérent
        total_text = " ".join([w["text"] for w in total_words])
        total_value = self.validator.normalize_amount(total_text, "total_line")
        
        # Validation croisée: total ≈ quantité * prix unitaire
        if unit_price and quantity and not total_value:
            total_value = round(unit_price * quantity, 2)
        elif total_value and quantity and not unit_price and quantity != 0:
            unit_price = round(total_value / quantity, 2)
        
        return {
            "quantity": int(quantity) if quantity else 1,
            "unit": self._detect_unit(quantity_words),
            "product_code": self._extract_product_code(desc_words),
            "description": description,
            "ean": None,  # Extraction spécifique si barcode présent
            "unit_price": unit_price,
            "total_value": total_value
        }
    
    def _extract_product_code(self, words: List[Dict]) -> Optional[str]:
        """Extrait un code produit des mots de description"""
        text = " ".join([w["text"] for w in words])
        
        # Codes produits: souvent en début de description, alphanumériques
        parts = text.split()
        for part in parts[:2]:  # Chercher dans les 2 premiers mots
            # Code produit typique: ABC-123, REF123, etc.
            if any(c.isdigit() for c in part) and any(c.isalpha() for c in part):
                if 5 <= len(part) <= 20:
                    return part
            # Ou purement numérique mais long (SKU)
            elif part.isdigit() and 6 <= len(part) <= 13:
                return part
        
        return None
    
    def _detect_unit(self, words: List[Dict]) -> Optional[str]:
        """Détecte l'unité de mesure"""
        text = " ".join([w["text_lower"] for w in words])
        
        units = {
            "pcs": ["pcs", "pce", "pc", "piece", "pieces", "stück", "stck"],
            "kg": ["kg", "kgs", "kilos", "kilogramme"],
            "g": ["g", "gr", "gramme", "grammes"],
            "l": ["l", "lt", "liter", "litre", "litres"],
            "ml": ["ml", "milliliter"],
            "m": ["m", "meter", "metre", "mètre"],
            "m²": ["m2", "m²", "sqm", "m²"],
            "box": ["box", "carton", "ct", "colis"],
            "hour": ["h", "hour", "heure", "hr"],
            "day": ["day", "jour", "d", "tag"]
        }
        
        for unit, variants in units.items():
            if any(v in text for v in variants):
                return unit
        
        return "pcs"  # Défaut
    
    def extract_summary(self) -> Dict:
        """Extraction des totaux avec validation"""
        # Total TTC
        config_ttc = KEYWORDS["fields"]["montant_total"]
        total_ttc_str = self._find_field_with_context(config_ttc)
        total_ttc = self.validator.normalize_amount(total_ttc_str, "global_total")
        
        # Total HT
        config_ht = KEYWORDS["fields"]["montant_ht"]
        total_ht_str = self._find_field_with_context(config_ht)
        total_ht = self.validator.normalize_amount(total_ht_str, "global_total")
        
        # Calcul TVA si possible
        total_tva = None
        if total_ttc and total_ht:
            total_tva = round(total_ttc - total_ht, 2)
            taux = round((total_tva / total_ht) * 100, 1) if total_ht > 0 else None
        else:
            taux = None
        
        return {
            "tax_rate_total": f"{taux}%" if taux else None,
            "total_hors_taxe": total_ht,
            "total_tva": total_tva,
            "total_all_inclusive": total_ttc
        }
    
    def _extract_from_zone(self, zone: str) -> Optional[str]:
        """Extrait le texte d'une zone géographique"""
        page_rect = self.page.rect
        w, h = page_rect.width, page_rect.height
        
        zones = {
            "top_left": (0, 0, w*0.45, h*0.35),
            "top_right": (w*0.55, 0, w, h*0.35),
            "bottom_right": (w*0.5, h*0.65, w, h),
            "bottom_left": (0, h*0.65, w*0.5, h)
        }
        
        if zone not in zones:
            return None
        
        x0, y0, x1, y1 = zones[zone]
        words_in_zone = [
            w for w in self.words 
            if x0 <= w["x_center"] <= x1 and y0 <= w["y_center"] <= y1
        ]
        
        if not words_in_zone:
            return None
        
        words_in_zone.sort(key=lambda w: (w["y_center"], w["x_center"]))
        return " ".join([w["text"] for w in words_in_zone[:15]])
    
    def extract(self) -> Dict:
        """Méthode principale d'extraction"""
        doc_type = self.detect_document_type()
        
        return {
            "typeDocument": doc_type.value if doc_type != DocumentType.UNKNOWN else None,
            "numero": self.extract_invoice_number(),
            "client": self._find_field_with_context(KEYWORDS["fields"]["client"]),
            "dateDocument": self.extract_date(),
            "supplier": self._find_field_with_context(KEYWORDS["fields"]["supplier"]),
            "products": self.extract_products(),
            "summary": self.extract_summary()
        }

def process_document(pdf_path: str) -> str:
    extractor = RobustPDFExtractor(pdf_path)
    data = extractor.extract()
    return json.dumps(data, indent=2, ensure_ascii=False, default=str)
