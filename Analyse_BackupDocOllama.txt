Qu'est-ce que LLooMA ?
LLooMA (LLM Orchestration & Management Architecture) est un système d'orchestration network-native qui opère comme une couche de contrôle au-dessus des modèles de langage traditionnels. Contrairement à Ollama ou vLLM qui sont des runtimes d'inférence, LLooMA n'existe pas comme un ensemble de poids ou un artefact déployé sur une machine — il vit entièrement dans l'orchestrateur PeerLLM.

Cycle de vie d'une requête dans LLooMA
Table
Phase	Description	Technique clé
1. Validation	Auth + solde tokens	Early gating économique
2. Analyse	Déterminer si décomposition nécessaire	Heuristiques (longueur, complexité)
3. Décomposition	Transformation en DAG de tâches	Graphe acyclique dirigé
4. Exécution	Course dual-host par tâche	Parallelisme + redondance
5. Normalisation	Standardisation des outputs	Compression de contexte
6. Agrégation	Fusion en réponse cohérente	Stratégies de merge

// PeerLLMClient.cs
public class PeerLLMClient
{
    private readonly HttpClient _httpClient;
    private readonly string _apiKey;
    
    public PeerLLMClient(string baseUrl, string apiKey)
    {
        _apiKey = apiKey;
        _httpClient = new HttpClient 
        { 
            BaseAddress = new Uri(baseUrl) // https://api.peerllm.com
        };
        _httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {_apiKey}");
    }

    /// <summary>
    /// Envoie une requête à LLooMA qui sera orchestrée automatiquement
    /// </summary>
    public async Task<LLooMAResponse> GenerateAsync(
        string prompt, 
        string? model = null,
        LLooMAOptions? options = null)
    {
        var request = new
        {
            prompt = prompt,
            model = model, // null = choix auto par LLooMA
            options = options ?? new LLooMAOptions
            {
                MaxLatency = TimeSpan.FromSeconds(30),
                EnableDecomposition = true, // Permet à LLooMA de décomposer
                ParallelExecution = true,
                ContextCompression = true
            }
        };

        var response = await _httpClient.PostAsJsonAsync("/v1/orchestrate", request);
        return await response.Content.ReadFromJsonAsync<LLooMAResponse>();
    }
}

public class LLooMAOptions
{
    public TimeSpan MaxLatency { get; set; }
    public bool EnableDecomposition { get; set; } // Active le DAG
    public bool ParallelExecution { get; set; }   // Course multi-hosts
    public bool ContextCompression { get; set; }  // Compression contexte
}

public class LLooMAResponse
{
    public string Content { get; set; }
    public LLooMAMetadata Metadata { get; set; }
}

public class LLooMAMetadata
{
    public int TasksExecuted { get; set; }      // Nombre de tâches dans le DAG
    public int HostsUsed { get; set; }          // Hosts participants
    public TimeSpan Latency { get; set; }       // Latence totale
    public string ExecutionStrategy { get; set; } // "fast_path" ou "decomposed"
}

// LLooMAStreamingService.cs
public class LLooMAStreamingService
{
    private readonly HttpClient _httpClient;
    
    public async IAsyncEnumerable<string> StreamGenerateAsync(
        string prompt,
        [EnumeratorCancellation] CancellationToken ct = default)
    {
        var request = new { prompt = prompt, stream = true };
        
        using var response = await _httpClient.PostAsJsonAsync(
            "/v1/orchestrate/stream", 
            request, 
            cancellationToken: ct);
        
        using var stream = await response.Content.ReadAsStreamAsync();
        using var reader = new StreamReader(stream);
        
        while (!reader.EndOfStream && !ct.IsCancellationRequested)
        {
            var line = await reader.ReadLineAsync();
            if (line?.StartsWith("data: ") == true)
            {
                var json = line[6..];
                var chunk = JsonSerializer.Deserialize<StreamChunk>(json);
                yield return chunk.Content;
            }
        }
    }
}

// llm.service.ts
import { Injectable } from '@angular/core';
import { HttpClient, HttpEventType } from '@angular/common/http';
import { Observable, Subject } from 'rxjs';
import { map } from 'rxjs/operators';

interface ChatRequest {
  message: string;
  model?: string;
  timeoutSeconds?: number;
}

interface ChatResponse {
  message: string;
  orchestration: 'fast_path' | 'decomposed';
  latency: number;
}

@Injectable({ providedIn: 'root' })
export class LLooMAService {
  private apiUrl = 'http://localhost:5000/api';

  constructor(private http: HttpClient) {}

  // Requête standard (LLooMA décide de la stratégie)
  chat(request: ChatRequest): Observable<ChatResponse> {
    return this.http.post<ChatResponse>(`${this.apiUrl}/chat`, request);
  }

  // Streaming pour UX temps réel
  streamChat(message: string): Observable<string> {
    return new Observable(observer => {
      const eventSource = new EventSource(
        `${this.apiUrl}/chat/stream?message=${encodeURIComponent(message)}`
      );

      eventSource.onmessage = (event) => {
        observer.next(event.data);
      };

      eventSource.onerror = (error) => {
        observer.error(error);
        eventSource.close();
      };

      return () => eventSource.close();
    });
  }
}

          {{ loading ? 'Traitement...' : 'Envoyer' }}
        </button>
      </div>
    </div>
  `,
  styles: [`
    .strategy-badge { padding: 4px 8px; border-radius: 4px; font-size: 12px; }
    .fast_path { background: #4caf50; color: white; }
    .decomposed { background: #2196f3; color: white; }
    .streaming-indicator { animation: pulse 1s infinite; }
  `]
})
export class ChatComponent {
  messages: { role: 'user' | 'assistant', content: string }[] = [];
  userInput = '';
  loading = false;
  streaming = false;
  currentStrategy: string | null = null;
  latency = 0;

  constructor(private llmService: LLooMAService) {}

  sendMessage() {
    if (!this.userInput.trim()) return;

    this.messages.push({ role: 'user', content: this.userInput });
    this.loading = true;
    const userMsg = this.userInput;
    this.userInput = '';

    // Utiliser le streaming pour meilleure UX
    this.streaming = true;
    let assistantContent = '';

    this.llmService.streamChat(userMsg).subscribe({
      next: (chunk) => {
        assistantContent += chunk;
        // Mettre à jour le dernier message ou créer
        if (this.messages[this.messages.length - 1].role === 'assistant') {
          this.messages[this.messages.length - 1].content = assistantContent;
        } else {
          this.messages.push({ role: 'assistant', content: assistantContent });
        }
      },
      error: (err) => {
        console.error('Erreur LLooMA:', err);
        this.loading = false;
        this.streaming = false;
      },
      complete: () => {
        this.loading = false;
        this.streaming = false;
        // Optionnel: récupérer les métadonnées via endpoint non-streaming
      }
    });
  }
}

public enum TypeDocument { BonLivraison, Facture }

public class Document
{
    public Guid Id { get; set; }
    public TypeDocument Type { get; set; }
    public string Numero { get; set; } = null!;
    public DateTime Date { get; set; }
    public string Fournisseur { get; set; } = null!;
    public List<LigneDocument> Lignes { get; set; } = new();
    public string? CheminFichier { get; set; }
    public DateTime DateExtraction { get; set; }
}

public class LigneDocument
{
    public string Reference { get; set; } = null!;
    public string Designation { get; set; } = null!;
    public decimal Quantite { get; set; }
    public decimal PrixUnitaire { get; set; }
    public decimal TotalLigne => Quantite * PrixUnitaire;
}

public class EcartReconciliation
{
    public Guid Id { get; set; }
    public string Reference { get; set; } = null!;
    public string Designation { get; set; } = null!;
    
    // Valeurs BL
    public decimal QteBL { get; set; }
    public decimal PrixUnitaireBL { get; set; }
    
    // Valeurs Facture
    public decimal QteFacture { get; set; }
    public decimal PrixUnitaireFacture { get; set; }
    
    // Écarts calculés
    public decimal EcartQuantite => QteFacture - QteBL;
    public decimal EcartPrixUnitaire => PrixUnitaireFacture - PrixUnitaireBL;
    public decimal EcartTotal => (QteFacture * PrixUnitaireFacture) - (QteBL * PrixUnitaireBL);
    
    public TypeEcart Type { get; set; }
    public StatutEcart Statut { get; set; }
    public string? Commentaire { get; set; }
}

public enum TypeEcart { Quantite, Prix, QuantiteEtPrix, ReferenceInconnue }
public enum StatutEcart { Detecte, Valide, Corrige, Ignore }

// Services/LLooMAExtractionService.cs
public class LLooMAExtractionService
{
    private readonly PeerLLMClient _peerLLM;
    private readonly ILogger<LLooMAExtractionService> _logger;

    public LLooMAExtractionService(PeerLLMClient peerLLM, ILogger<LLooMAExtractionService> logger)
    {
        _peerLLM = peerLLM;
        _logger = logger;
    }

    public async Task<Document> ExtraireDocumentAsync(string texteOCR, TypeDocument type)
    {
        var prompt = type == TypeDocument.BonLivraison 
            ? ConstruirePromptBL(texteOCR)
            : ConstruirePromptFacture(texteOCR);

        var response = await _peerLLM.GenerateAsync(prompt, options: new LLooMAOptions
        {
            EnableDecomposition = true, // Pour documents complexes/multi-pages
            MaxLatency = TimeSpan.FromSeconds(45)
        });

        return ParserReponseLLooMA(response.Content, type);
    }

    private string ConstruirePromptBL(string texteOCR) => $@"
Tu es un extracteur de données structurées pour documents commerciaux.
Analyse ce texte OCR d'un BON DE LIVRAISON et extrais les informations au format JSON strict.

Texte OCR:
---
{texteOCR}
---

Règles d'extraction:
1. Numéro de document: cherche 'BL', 'Bon de livraison', 'Livraison n°'
2. Date: format ISO 8601 (YYYY-MM-DD)
3. Fournisseur: nom de l'entreprise émettrice
4. Lignes: tableau d'articles avec:
   - reference: code article (souvent alphanumérique)
   - designation: description produit
   - quantite: nombre (décimal si besoin)
   - prix_unitaire: prix HT unitaire

Réponds UNIQUEMENT avec ce JSON, sans markdown:
{{
  ""numero"": ""string"",
  ""date"": ""YYYY-MM-DD"",
  ""fournisseur"": ""string"",
  ""lignes"": [
    {{
      ""reference"": ""string"",
      ""designation"": ""string"",
      ""quantite"": 0.00,
      ""prix_unitaire"": 0.00
    }}
  ]
}}";

    private string ConstruirePromptFacture(string texteOCR) => $@"
Tu es un extracteur de données structurées pour documents commerciaux.
Analyse ce texte OCR d'une FACTURE et extrais les informations au format JSON strict.

Texte OCR:
---
{texteOCR}
---

Règles spécifiques facture:
1. Numéro: cherche 'Facture n°', 'FA', 'Invoice'
2. Référence BL associée: souvent mentionnée comme 'Votre BL', 'Réf BL', 'Livraison'
3. Prix: peuvent être TTC ou HT - précise dans le JSON si indiqué

Réponds UNIQUEMENT avec ce JSON:
{{
  ""numero"": ""string"",
  ""date"": ""YYYY-MM-DD"",
  ""fournisseur"": ""string"",
  ""reference_bl"": ""string"",
  ""lignes"": [
    {{
      ""reference"": ""string"",
      ""designation"": ""string"",
      ""quantite"": 0.00,
      ""prix_unitaire"": 0.00,
      ""type_prix"": ""HT|TTC""
    }}
  ]
}}";

    private Document ParserReponseLLooMA(string json, TypeDocument type)
    {
        try
        {
            using var doc = JsonDocument.Parse(json);
            var root = doc.RootElement;
            
            var document = new Document
            {
                Id = Guid.NewGuid(),
                Type = type,
                Numero = root.GetProperty("numero").GetString()!,
                Date = DateTime.Parse(root.GetProperty("date").GetString()!),
                Fournisseur = root.GetProperty("fournisseur").GetString()!,
                DateExtraction = DateTime.UtcNow
            };

            foreach (var ligne in root.GetProperty("lignes").EnumerateArray())
            {
                document.Lignes.Add(new LigneDocument
                {
                    Reference = ligne.GetProperty("reference").GetString()!,
                    Designation = ligne.GetProperty("designation").GetString()!,
                    Quantite = ligne.GetProperty("quantite").GetDecimal(),
                    PrixUnitaire = ligne.GetProperty("prix_unitaire").GetDecimal()
                });
            }

            return document;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Erreur parsing réponse LLooMA");
            throw new InvalidOperationException("Format de réponse LLooMA invalide", ex);
        }
    }
}

// Services/ReconciliationService.cs
public class ReconciliationService
{
    public List<EcartReconciliation> Reconcilier(Document bl, Document facture)
    {
        var ecarts = new List<EcartReconciliation>();
        
        // Indexer les lignes par référence
        var lignesBL = bl.Lignes.ToDictionary(l => NormaliserReference(l.Reference));
        var lignesFacture = facture.Lignes.ToDictionary(l => NormaliserReference(l.Reference));

        // 1. Vérifier chaque ligne de la facture
        foreach (var (refFact, ligneFact) in lignesFacture)
        {
            if (!lignesBL.TryGetValue(refFact, out var ligneBL))
            {
                // Référence inconnue dans le BL
                ecarts.Add(new EcartReconciliation
                {
                    Id = Guid.NewGuid(),
                    Reference = ligneFact.Reference,
                    Designation = ligneFact.Designation,
                    QteBL = 0,
                    PrixUnitaireBL = 0,
                    QteFacture = ligneFact.Quantite,
                    PrixUnitaireFacture = ligneFact.PrixUnitaire,
                    Type = TypeEcart.ReferenceInconnue,
                    Statut = StatutEcart.Detecte
                });
                continue;
            }

            // Calculer écarts
            var ecartQte = ligneFact.Quantite - ligneBL.Quantite;
            var ecartPrix = ligneFact.PrixUnitaire - ligneBL.PrixUnitaire;

            if (ecartQte != 0 || ecartPrix != 0)
            {
                var type = ecartQte != 0 && ecartPrix != 0 
                    ? TypeEcart.QuantiteEtPrix
                    : ecartQte != 0 
                        ? TypeEcart.Quantite 
                        : TypeEcart.Prix;

                ecarts.Add(new EcartReconciliation
                {
                    Id = Guid.NewGuid(),
                    Reference = ligneFact.Reference,
                    Designation = ligneFact.Designation,
                    QteBL = ligneBL.Quantite,
                    PrixUnitaireBL = ligneBL.PrixUnitaire,
                    QteFacture = ligneFact.Quantite,
                    PrixUnitaireFacture = ligneFact.PrixUnitaire,
                    Type = type,
                    Statut = StatutEcart.Detecte
                });
            }
        }

        // 2. Vérifier les lignes BL manquantes dans la facture
        foreach (var (refBL, ligneBL) in lignesBL)
        {
            if (!lignesFacture.ContainsKey(refBL))
            {
                ecarts.Add(new EcartReconciliation
                {
                    Id = Guid.NewGuid(),
                    Reference = ligneBL.Reference,
                    Designation = ligneBL.Designation,
                    QteBL = ligneBL.Quantite,
                    PrixUnitaireBL = ligneBL.PrixUnitaire,
                    QteFacture = 0,
                    PrixUnitaireFacture = 0,
                    Type = TypeEcart.Quantite, // Manquant = qte 0
                    Statut = StatutEcart.Detecte,
                    Commentaire = "Ligne BL absente de la facture"
                });
            }
        }

        return ecarts;
    }

    private string NormaliserReference(string reference) => 
        reference.Trim().ToUpperInvariant().Replace(" ", "");
}

// Services/StockService.cs
public class StockService
{
    private readonly ApplicationDbContext _context;
    private readonly ILogger<StockService> _logger;

    public async Task<ResultatMajStock> MettreAJourStockAsync(
        List<EcartReconciliation> ecartsValides,
        string numeroBL)
    {
        var resultat = new ResultatMajStock();
        
        using var transaction = await _context.Database.BeginTransactionAsync();

        try
        {
            foreach (var ecart in ecartsValides.Where(e => e.Statut == StatutEcart.Valide))
            {
                // Rechercher l'article
                var article = await _context.Articles
                    .FirstOrDefaultAsync(a => a.Reference == ecart.Reference);

                if (article == null)
                {
                    resultat.Erreurs.Add($"Article {ecart.Reference} non trouvé");
                    continue;
                }

                // Déterminer la quantité à prendre en compte
                var qteFinale = ecart.QteFacture; // Par défaut, on prend la facture
                
                // Si l'utilisateur a corrigé, prendre la valeur corrigée
                if (ecart.Type == TypeEcart.Quantite || ecart.Type == TypeEcart.QuantiteEtPrix)
                {
                    // Logique: si BL > Facture, il manque des pièces
                    // Si Facture > BL, excédent (à vérifier avec fournisseur)
                    
                    var mouvement = new MouvementStock
                    {
                        Id = Guid.NewGuid(),
                        ArticleId = article.Id,
                        Type = TypeMouvement.Entree,
                        Quantite = qteFinale,
                        ReferenceDocument = numeroBL,
                        Date = DateTime.UtcNow,
                        Motif = $"Réconciliation BL/Facture - Écart: {ecart.EcartQuantite}"
                    };

                    _context.MouvementsStock.Add(mouvement);
                    
                    // MàJ quantité stock
                    article.QuantiteDisponible += qteFinale;
                    
                    resultat.MouvementsCrees.Add(mouvement);
                }

                // Noter l'écart de prix pour analyse fournisseur
                if (ecart.EcartPrixUnitaire != 0)
                {
                    _context.EcartsPrixFournisseur.Add(new EcartPrixFournisseur
                    {
                        Fournisseur = ecart.Designation,
                        Reference = ecart.Reference,
                        PrixAttendu = ecart.PrixUnitaireBL,
                        PrixFacture = ecart.PrixUnitaireFacture,
                        DateDetection = DateTime.UtcNow
                    });
                }
            }

            await _context.SaveChangesAsync();
            await transaction.CommitAsync();
            
            resultat.Succes = true;
            return resultat;
        }
        catch (Exception ex)
        {
            await transaction.RollbackAsync();
            _logger.LogError(ex, "Erreur mise à jour stock");
            resultat.Erreurs.Add(ex.Message);
            return resultat;
        }
    }
}

    ecarts: Ecart[],

  ): Observable<any> {
    return this.http.post(`${this.apiUrl}/valider-et-mettre-a-jour-stock`, {
      ecarts,
      numeroBL
    });
  }

// components/reconciliation/reconciliation.component.ts
@Component({
  selector: 'app-reconciliation',
  templateUrl: './reconciliation.component.html',
  styleUrls: ['./reconciliation.component.scss']
})
export class ReconciliationComponent {
  blFile: File | null = null;
  factureFile: File | null = null;
  documentBL: Document | null = null;
  documentFacture: Document | null = null;
  reconciliationResult: ReconciliationResult | null = null;
  loading = false;
  step: 'upload' | 'review' | 'ecarts' | 'validation' = 'upload';

  constructor(private reconciliationService: ReconciliationService) {}

  onUploadBL(event: any) {
    this.blFile = event.target.files[0];
    this.extraireDocument(this.blFile!, 'BonLivraison', 'bl');
  }

  onUploadFacture(event: any) {
    this.factureFile = event.target.files[0];
    this.extraireDocument(this.factureFile!, 'Facture', 'facture');
  }

  private extraireDocument(
    fichier: File, 
    type: 'BonLivraison' | 'Facture',
    cible: 'bl' | 'facture'
  ) {
    this.loading = true;
    this.reconciliationService.uploadAndExtract(fichier, type)
      .subscribe({
        next: (doc) => {
          if (cible === 'bl') this.documentBL = doc;
          else this.documentFacture = doc;
          this.loading = false;
          
          if (this.documentBL && this.documentFacture) {
            this.comparer();
          }
        },
        error: (err) => {
          this.loading = false;
          alert('Erreur extraction: ' + err.message);
        }
      });
  }

  comparer() {
    if (!this.documentBL || !this.documentFacture) return;
    
    this.loading = true;
    this.reconciliationService.comparerDocuments(this.documentBL, this.documentFacture)
      .subscribe({
        next: (result) => {
          this.reconciliationResult = result;
          this.step = result.ecarts.length > 0 ? 'ecarts' : 'validation';
          this.loading = false;
        },
        error: (err) => {
          this.loading = false;
          alert('Erreur comparaison: ' + err.error.erreur);
        }
      });
  }

  validerEcart(ecart: Ecart, action: 'Valide' | 'Corrige' | 'Ignore') {
    ecart.statut = action;
    
    if (action === 'Corrige') {
      // Ouvrir modal de correction
      this.ouvrirModalCorrection(ecart);
    }
  }

  corrigerEcart(ecart: Ecart, nouvelleQte: number, nouveauPrix: number) {
    ecart.qteFacture = nouvelleQte;
    ecart.prixUnitaireFacture = nouveauPrix;
    ecart.statut = 'Corrige';
  }

  peutValiderStock(): boolean {
    return this.reconciliationResult?.ecarts.every(
      e => e.statut !== 'Detecte'
    ) ?? false;
  }

  validerEtMettreAJourStock() {
    if (!this.peutValiderStock()) return;
    
    this.loading = true;
    this.reconciliationService.validerEtMettreAJourStock(
      this.reconciliationResult!.ecarts,
      this.documentBL!.numero
    ).subscribe({
      next: (result) => {
        this.loading = false;
        alert(`Stock mis à jour! ${result.mouvementsCrees?.length || 0} mouvements créés.`);
        this.step = 'validation';
      },
      error: (err) => {
        this.loading = false;
        alert('Erreur mise à jour stock: ' + err.message);
      }
    });
  }
}

// Services/OllamaLocalService.cs
public class OllamaLocalService
{
    private readonly HttpClient _httpClient;
    private readonly ILogger<OllamaLocalService> _logger;
    private readonly OllamaConfig _config;

    public OllamaLocalService(HttpClient httpClient, IOptions<OllamaConfig> config, ILogger<OllamaLocalService> logger)
    {
        _httpClient = httpClient;
        _config = config.Value;
        _httpClient.BaseAddress = new Uri(_config.BaseUrl);
        _logger = logger;
    }

    /// <summary>
    /// Extrait les données structurées d'un document avec détection langue
    /// </summary>
    public async Task<ExtractionResult> ExtraireDocumentAsync(
        string texteOCR, 
        TypeDocument type,
        string? langueDetectee = null)
    {
        // Détection langue si non fournie
        langueDetectee ??= DetecterLangue(texteOCR);
        
        // Sélection modèle adapté
        var model = _config.ModelsByLanguage.GetValueOrDefault(langueDetectee, "llama3.1:8b");
        
        _logger.LogInformation("Extraction {Type} en [{Langue}] avec modèle {Model}", 
            type, langueDetectee, model);

        var prompt = ConstruirePromptMultilingue(texteOCR, type, langueDetectee);
        
        var request = new
        {
            model = model,
            prompt = prompt,
            stream = false,
            options = new
            {
                temperature = 0.1,  // Précision maximale
                num_predict = 2000  // Réponse longue pour JSON complet
            }
        };

        try
        {
            var response = await _httpClient.PostAsJsonAsync("/api/generate", request);
            response.EnsureSuccessStatusCode();
            
            var result = await response.Content.ReadFromJsonAsync<OllamaResponse>();
            return ParserReponse(result!.Response, type, langueDetectee);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Erreur Ollama pour {Langue}", langueDetectee);
            throw;
        }
    }

    private string DetecterLangue(string texte)
    {
        // Utilisation de langdetect ou heuristic simple
        if (texte.Contains("Bon de livraison") || texte.Contains("Facture")) return "fr";
        if (texte.Contains("Delivery Note") || texte.Contains("Invoice")) return "en";
        if (texte.Contains("Lieferschein") || texte.Contains("Rechnung")) return "de";
        if (texte.Contains("Nota de entrega") || texte.Contains("Factura")) return "es";
        
        // Fallback: détection par caractères
        if (texte.Any(c => c >= 0x4E00 && c <= 0x9FFF)) return "zh"; // Chinois
        if (texte.Any(c => c >= 0x3040 && c <= 0x309F)) return "ja"; // Hiragana japonais
        
        return "default";
    }

    private string ConstruirePromptMultilingue(string texteOCR, TypeDocument type, string langue)
    {
        var instructions = langue switch
        {
            "fr" => @"
Tu es un extracteur de données pour documents commerciaux.
Extrais les informations de ce document au format JSON strict.
Règles:
- 'numero': numéro de document (BL, Facture, Invoice...)
- 'date': format ISO YYYY-MM-DD
- 'fournisseur': nom société émettrice
- 'lignes': tableau articles avec reference, designation, quantite, prix_unitaire_HT
- Si prix TTC mentionné, calculer prix HT avec taux TVA standard du pays",
            
            "en" => @"
You are a data extractor for commercial documents.
Extract information to strict JSON format.
Rules:
- 'numero': document number (Delivery Note, Invoice, BL...)
- 'date': ISO format YYYY-MM-DD
- 'fournisseur': issuing company name
- 'lignes': array of items with reference, designation, quantite, prix_unitaire_HT

    private ExtractionResult ParserReponse(string json, TypeDocument type, string langue)
    {
        // Nettoyer le JSON (Ollama peut ajouter des ```json)
        json = json.Trim();
        if (json.StartsWith("```json")) json = json[7..];
        if (json.StartsWith("```")) json = json[3..];
        if (json.EndsWith("```")) json = json[..^3];
        json = json.Trim();

        try
        {
            using var doc = JsonDocument.Parse(json);
            var root = doc.RootElement;

            var result = new ExtractionResult
            {
                LangueDetectee = langue,
                Type = type,
                Numero = ExtraireChamp(root, "numero", "numéro", "number", "nummer"),
                Date = ParserDate(ExtraireChamp(root, "date")),
                Fournisseur = ExtraireChamp(root, "fournisseur", "supplier", "lieferant", "proveedor", "供应商"),
                Lignes = new List<LigneExtrait>()
            };

            // Gestion multilingue des champs de ligne
            var lignesArray = root.GetProperty("lignes");
            foreach (var ligne in lignesArray.EnumerateArray())
            {
                result.Lignes.Add(new LigneExtrait
                {
                    Reference = ExtraireChamp(ligne, "reference", "ref", "artikelnummer", "referencia", "参考"),
                    Designation = ExtraireChamp(ligne, "designation", "description", "bezeichnung", "descripcion", "描述"),
                    Quantite = ExtraireDecimal(ligne, "quantite", "quantity", "menge", "cantidad", "数量"),
                    PrixUnitaire = ExtraireDecimal(ligne, "prix_unitaire", "unit_price", "einzelpreis", "precio_unitario", "单价"),
                    TypePrix = ExtraireChampOptionnel(ligne, "type_prix") ?? "HT"
                });
            }

            // Si prix TTC, convertir en HT (taux par défaut selon pays)
            if (type == TypeDocument.Facture)
            {
                foreach (var ligne in result.Lignes.Where(l => l.TypePrix == "TTC"))
                {
                    var tauxTVA = GetTauxTVA(langue);
                    ligne.PrixUnitaire = ligne.PrixUnitaire / (1 + tauxTVA);
                    ligne.TypePrix = "HT";
                }
            }

            return result;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Erreur parsing JSON Ollama: {Json}", json);
            throw new InvalidOperationException($"Réponse Ollama invalide: {ex.Message}");
        }
    }

    private string ExtraireChamp(JsonElement element, params string[] nomsPossibles)
    {
        foreach (var nom in nomsPossibles)
        {
            if (element.TryGetProperty(nom, out var prop))
                return prop.GetString() ?? "";
        }
        return "";
    }

    private decimal ExtraireDecimal(JsonElement element, params string[] nomsPossibles)
    {
        foreach (var nom in nomsPossibles)
        {
            if (element.TryGetProperty(nom, out var prop))
            {
                if (prop.ValueKind == JsonValueKind.Number)
                    return prop.GetDecimal();
                if (prop.ValueKind == JsonValueKind.String && decimal.TryParse(prop.GetString(), out var val))
                    return val;
            }
        }
        return 0;
    }

    private decimal GetTauxTVA(string langue) => langue switch
    {
        "fr" => 0.20m,  // France 20%
        "de" => 0.19m,  // Allemagne 19%
        "es" => 0.21m,  // Espagne 21%
        "it" => 0.22m,  // Italie 22%
        "en" => 0.20m,  // UK 20% (ou 0 si hors EU)
        _ => 0.20m
    };
}

public class OCRMultilingueService
{
    private readonly ILogger<OCRMultilingueService> _logger;
    
    // Mapping langue -> traineddata Tesseract
    private static readonly Dictionary<string, string[]> LanguesTesseract = new()
    {
        ["fr"] = new[] { "fra", "eng" },  // Français + Anglais (pour références)
        ["en"] = new[] { "eng" },
        ["de"] = new[] { "deu", "eng" },
        ["es"] = new[] { "spa", "eng" },
        ["it"] = new[] { "ita", "eng" },
        ["zh"] = new[] { "chi_sim", "chi_tra", "eng" },
        ["ja"] = new[] { "jpn", "eng" },
        ["default"] = new[] { "eng", "fra", "deu" } // Toutes les langues EU
    };

    public async Task<OCRResult> ExtraireTexteAsync(Stream fichier, string extension)
    {
        // Convertir en images si PDF
        var images = extension.ToLower() switch
        {
            ".pdf" => ConvertirPdfEnImages(fichier),
            _ => new[] { ChargerImage(fichier) }
        };

        var texteComplet = new StringBuilder();
        var languesUtilisees = new HashSet<string>();

        foreach (var image in images)
        {
            // Essayer plusieurs langues et garder le meilleur résultat
            var meilleurResultat = await EssayerLanguesAsync(image);
            texteComplet.AppendLine(meilleurResultat.Texte);
            languesUtilisees.Add(meilleurResultat.Langue);
        }

        // Détection finale langue dominante
        var langueDominante = DetecterLangueDominante(texteComplet.ToString());

        return new OCRResult
        {
            Texte = texteComplet.ToString(),
            LangueDetectee = langueDominante,
            LanguesEssayees = languesUtilisees.ToList(),
            NombrePages = images.Length
        };
    }

    private async Task<(string Texte, string Langue)> EssayerLanguesAsync(Bitmap image)
    {
        string meilleurTexte = "";
        string meilleureLangue = "eng";
        double meilleureConfiance = 0;

        foreach (var langue in new[] { "fra", "eng", "deu", "spa" }) // Ordre priorité EU
        {
            try
            {
                using var engine = new TesseractEngine(
                    @"./tessdata", 
                    langue, 
                    EngineMode.Default);
                
                engine.SetVariable("tessedit_char_whitelist", 
                    "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz.,;:/-_%€$£¥ ");

                using var img = PixConverter.ToPix(image);
                using var page = engine.Process(img);
                
                var confiance = page.GetMeanConfidence();
                var texte = page.GetText();

                // Heuristique: si beaucoup de chiffres et références, c'est probablement bon
                if (confiance > meilleureConfiance && texte.Length > 50)
                {
                    meilleureConfiance = confiance;
                    meilleurTexte = texte;
                    meilleureLangue = langue;
                }
            }
            catch { /* Ignorer erreur langue non installée */ }
        }

        return (meilleurTexte, meilleureLangue);
    }

    private string DetecterLangueDominante(string texte)
    {
        // Comptage mots clés par langue
        var scores = new Dictionary<string, int>
        {
            ["fr"] = CompterOccurrences(texte, "facture", "bon", "livraison", "quantité", "prix", "total", "tva"),
            ["en"] = CompterOccurrences(texte, "invoice", "delivery", "quantity", "price", "total", "vat"),
            ["de"] = CompterOccurrences(texte, "rechnung", "lieferschein", "menge", "preis", "gesamt", "mwst"),
            ["es"] = CompterOccurrences(texte, "factura", "albarán", "cantidad", "precio", "total", "iva"),
            ["it"] = CompterOccurrences(texte, "fattura", "ddt", "quantità", "prezzo", "totale", "iva")
        };

        return scores.OrderByDescending(s => s.Value).First().Key;
    }

        mots.Count(m => texte.IndexOf(m, StringComparison.OrdinalIgnoreCase) >= 0);

[ApiController]
[Route("api/[controller]")]
public class ReconciliationOllamaController : ControllerBase
{
    private readonly OCRMultilingueService _ocrService;
    private readonly OllamaLocalService _ollamaService;
    private readonly ReconciliationService _reconciliationService;
    private readonly StockService _stockService;
    private readonly ILogger<ReconciliationOllamaController> _logger;

    [HttpPost("extract")]
    [Consumes("multipart/form-data")]
    public async Task<ActionResult<ExtractionResult>> ExtractDocument(
        IFormFile fichier,
        [FromQuery] TypeDocument type)
    {
        if (fichier == null || fichier.Length == 0)
            return BadRequest("Fichier requis");

        var extension = Path.GetExtension(fichier.FileName);
        
        try
        {
            // 1. OCR avec détection langue
            using var stream = fichier.OpenReadStream();
            var ocrResult = await _ocrService.ExtraireTexteAsync(stream, extension);
            
            _logger.LogInformation("OCR terminé - Langue: {Langue}, Caractères: {Length}", 
                ocrResult.LangueDetectee, ocrResult.Texte.Length);

            // 2. Extraction structurée via Ollama Local
            var extraction = await _ollamaService.ExtraireDocumentAsync(
                ocrResult.Texte, 
                type, 
                ocrResult.LangueDetectee);

            // 3. Sauvegarder temporairement en session/cache
            var docId = Guid.NewGuid().ToString();
            HttpContext.Session.SetString($"doc_{docId}", JsonSerializer.Serialize(extraction));

            return Ok(new
            {
                Id = docId,
                extraction.LangueDetectee,
                extraction.Type,
                extraction.Numero,
                extraction.Date,
                extraction.Fournisseur,
                extraction.Lignes,
                OcrPreview = ocrResult.Texte[..Math.Min(200, ocrResult.Texte.Length)] + "..."
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Erreur extraction {Fichier}", fichier.FileName);
            return StatusCode(500, new { erreur = ex.Message, details = ex.InnerException?.Message });
        }
    }

    [HttpPost("compare")]
    public async Task<ActionResult> CompareDocuments([FromBody] CompareRequest request)
    {
        var bl = JsonSerializer.Deserialize<ExtractionResult>(
            HttpContext.Session.GetString($"doc_{request.BlId}")!);
        var facture = JsonSerializer.Deserialize<ExtractionResult>(
            HttpContext.Session.GetString($"doc_{request.FactureId}")!);

        if (bl == null || facture == null)
            return BadRequest("Documents non trouvés en session");

        // Vérification cohérence
        if (!bl.Fournisseur.ToLower().Contains(facture.Fournisseur.ToLower()) &&
            !facture.Fournisseur.ToLower().Contains(bl.Fournisseur.ToLower()))
        {
            return BadRequest(new { 
                erreur = "Fournisseurs différents", 
                bl = bl.Fournisseur, 
                facture = facture.Fournisseur 
            });
        }

        // Réconciliation
        var ecarts = _reconciliationService.Reconcilier(
            MapperEnDocument(bl), 
            MapperEnDocument(facture));

        // Suggestion Ollama pour écarts complexes
        var suggestion = ecarts.Count > 0 
            ? await ObtenirSuggestionOllama(ecarts, bl.LangueDetectee)
            : null;

        return Ok(new
        {
            Bl = bl,
            Facture = facture,
            Ecarts = ecarts,
            Suggestion = suggestion,
            PeutValider = !ecarts.Any(e => e.Type == TypeEcart.ReferenceInconnue)
        });
    }

    private async Task<string> ObtenirSuggestionOllama(List<EcartReconciliation> ecarts, string langue)
    {
        var prompt = langue switch
        {
            "fr" => $"Analyse ces écarts et suggère: {JsonSerializer.Serialize(ecarts.Select(e => new { e.Reference, e.Type, e.EcartQuantite, e.EcartPrixUnitaire }))}. Action recommandée?",
            "en" => $"Analyze these discrepancies and recommend action: {JsonSerializer.Serialize(ecarts)}",
            "de" => $"Analysieren Sie diese Abweichungen: {JsonSerializer.Serialize(ecarts)}",
            _ => $"Analyze: {JsonSerializer.Serialize(ecarts)}"
        };

        var request = new
        {
            model = langue == "fr" ? "mistral:7b" : "llama3.1:8b",
            prompt = prompt,
            stream = false
        };

        var response = await _httpClient.PostAsJsonAsync("/api/generate", request);
        var result = await response.Content.ReadFromJsonAsync<OllamaResponse>();
        return result?.Response ?? "Aucune suggestion";
    }

    [HttpPost("validate-stock")]
    public async Task<ActionResult> ValidateAndUpdateStock([FromBody] ValidationRequest request)
    {
        // Récupérer documents
        var bl = JsonSerializer.Deserialize<ExtractionResult>(
            HttpContext.Session.GetString($"doc_{request.BlId}")!);

        // Validation écarts
        var ecartsNonTraites = request.Ecarts.Where(e => e.Statut == StatutEcart.Detecte).ToList();
        if (ecartsNonTraites.Any())
        {
            return BadRequest(new { 
                erreur = "Écarts non traités", 
                ecarts = ecartsNonTraites 
            });
        }

        // Mise à jour stock avec quantités corrigées
        var result = await _stockService.MettreAJourStockAsync(
            request.Ecarts, 
            bl!.Numero,
            bl.LangueDetectee);

        // Archiver la réconciliation
        await ArchiverReconciliation(bl, request.Ecarts, result);

        return Ok(new
        {
            Success = result.Succes,
            Mouvements = result.MouvementsCrees.Count,
            ValeurTotale = result.ValeurTotaleMouvements,
            Erreurs = result.Erreurs
        });
    }

      blId,
      factureId
    });
  }

    return this.http.post(`${this.apiUrl}/validate-stock`, {
      blId,
      ecarts
    });
  }
}

  blData: any;
  factureData: any;
  comparisonResult: any;
  loading = false;
  updating = false;
  stockUpdated = false;

  constructor(private service: OllamaReconciliationService) {}

  getFlag(langue: string): string {
    const flags: { [key: string]: string } = {
      'fr': '🇫🇷', 'en': '🇬🇧', 'de': '🇩🇪', 
      'es': '🇪🇸', 'it': '🇮🇹', 'zh': '🇨🇳', 'ja': '🇯🇵'
    };
    return flags[langue] || '🏳️';
  }

  onDrop(event: DragEvent, type: 'bl' | 'facture') {
    event.preventDefault();
    const file = event.dataTransfer?.files[0];
    if (file) this.uploadFile(file, type);
  }

  uploadFile(file: File, type: 'bl' | 'facture') {
    this.service.extractDocument(file, type === 'bl' ? 'BonLivraison' : 'Facture')
      .subscribe({
        next: (data) => {
          if (type === 'bl') this.blData = data;
          else this.factureData = data;
        },
        error: (err) => alert('Erreur extraction: ' + err.error?.erreur)
      });
  }

  comparer() {
    this.loading = true;
    this.service.compareDocuments(this.blData.id, this.factureData.id)
      .subscribe({
        next: (result) => {
          this.comparisonResult = result;
          this.loading = false;
        },
        error: (err) => {
          alert(err.error?.erreur || 'Erreur comparaison');
          this.loading = false;
        }
      });
  }

  calculerEcartCorrige(ecart: any) {
    if (ecart.qteCorrigee != null) {
      ecart.ecartQuantite = ecart.qteCorrigee - ecart.qteBL;
    }
  }

  tousEcartsTraites(): boolean {
    return this.comparisonResult?.ecarts?.every((e: any) => e.statut !== 'Detecte');
  }

  validerStock() {
    this.updating = true;
    this.service.validateAndUpdateStock(this.blData.id, this.comparisonResult.ecarts)
      .subscribe({
        next: () => {
          this.stockUpdated = true;
          this.updating = false;
        },
        error: (err) => {
          alert('Erreur mise à jour: ' + err.error?.erreur);
          this.updating = false;
        }
      });
  }
}

public class OllamaGemmaService
{
    private readonly HttpClient _httpClient;
    private readonly ILogger<OllamaGemmaService> _logger;

    // Configuration Gemma optimisée pour extraction
    private readonly Dictionary<string, GemmaConfig> _configByLang = new()
    {
        ["fr"] = new GemmaConfig { Model = "mistral:7b", Fallback = "gemma2:9b" }, // Mistral meilleur FR
        ["en"] = new GemmaConfig { Model = "gemma2:9b", Temperature = 0.0 },
        ["de"] = new GemmaConfig { Model = "gemma2:9b", Temperature = 0.0 },
        ["es"] = new GemmaConfig { Model = "gemma2:9b", Temperature = 0.0 },
        ["it"] = new GemmaConfig { Model = "gemma2:9b", Temperature = 0.0 },
        ["default"] = new GemmaConfig { Model = "gemma2:9b", Temperature = 0.1 }
    };

    public async Task<ExtractionResult> ExtraireAvecGemma(
        string texteOCR, 
        TypeDocument type,
        string langue)
    {
        var config = _configByLang.GetValueOrDefault(langue, _configByLang["default"]);
        
        // Gemma nécessite des prompts très structurés
        var prompt = ConstruirePromptGemma(texteOCR, type, langue);

        var request = new
        {
            model = config.Model,
            prompt = prompt,
            stream = false,
            options = new
            {
                temperature = config.Temperature, // 0 = déterministe pour extraction
                num_predict = 1500,
                top_p = 0.95,
                stop = new[] { "\n\n", "```" } // Arrêter proprement
            }
        };

        _logger.LogInformation("Extraction avec {Model} pour langue [{Langue}]", 
            config.Model, langue);

        var response = await _httpClient.PostAsJsonAsync("/api/generate", request);
        var result = await response.Content.ReadFromJsonAsync<OllamaResponse>();
        
        return ParserReponseGemma(result!.Response, type);
    }

    private string ConstruirePromptGemma(string texteOCR, TypeDocument type, string langue)
    {
        // Gemma fonctionne mieux avec des instructions très explicites et formatées
        
        var systemPrompt = langue switch
        {
            "fr" => @"Tu es un système d'extraction de données comptables. Tu dois extraire les informations d'un document commercial et répondre UNIQUEMENT au format JSON demandé, sans aucun texte additionnel.",
            "en" => @"You are an accounting data extraction system. Extract information from the commercial document and respond ONLY with the requested JSON format, no additional text.",
            "de" => @"Sie sind ein buchhaltungsdatenextraktionssystem...",
            _ => @"Extract data from commercial document. Respond ONLY with JSON."
        };

        var jsonTemplate = type == TypeDocument.BonLivraison
            ? @"{""numero"":""EX: BL-2024-001"",""date"":""YYYY-MM-DD"",""fournisseur"":""Nom société"",""lignes"":[{""reference"":""REF001"",""designation"":""Description article"",""quantite"":10,""prix_unitaire"":25.50}]}"

        return $@"{systemPrompt}

Document à analyser:
{texteOCR}

    private ExtractionResult ParserReponseGemma(string reponse, TypeDocument type)
    {
        // Gemma a tendance à ajouter des espaces ou formater différemment
        reponse = reponse.Trim();
        
        // Extraire JSON si entouré de texte
        var match = Regex.Match(reponse, @"\{[\s\S]*\}");
        if (match.Success)
        {
            reponse = match.Value;
        }

        // Nettoyer caractères problématiques
        reponse = reponse.Replace("'", "\"") // Gemma utilise parfois des quotes simples
                         .Replace("\n", " ")
                         .Replace("\r", "");

        try
        {
            return JsonSerializer.Deserialize<ExtractionResult>(reponse, new JsonSerializerOptions
            {
                PropertyNameCaseInsensitive = true,
                NumberHandling = JsonNumberHandling.AllowReadingFromString
            })!;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Erreur parsing Gemma. Réponse brute: {Reponse}", reponse);
            throw;
        }
    }

    private ExtractionResult ParserReponseGemma4(string reponse, TypeDocument type)
    {
        // Gemma 4 est très fiable pour le JSON structuré [^28^]
        // mais on nettoie quand même par sécurité
        
        reponse = reponse.Trim();
        
        // Extraire le bloc JSON
        var jsonMatch = Regex.Match(reponse, @"\{(?:[^{}]|(?<open>\{)|(?<-open>\}))+(?(open)(?!))\}");
        if (jsonMatch.Success)
        {
            reponse = jsonMatch.Value;
        }

        try
        {
            var options = new JsonSerializerOptions
            {
                PropertyNameCaseInsensitive = true,
                NumberHandling = JsonNumberHandling.AllowReadingFromString,
                AllowTrailingCommas = true // Gemma 4 peut ajouter des virgules
            };

            var result = JsonSerializer.Deserialize<ExtractionResult>(reponse, options);
            
            // Validation post-extraction
            if (string.IsNullOrEmpty(result?.Numero))
                throw new InvalidOperationException("Numéro de document non extrait");
            
            if (result.Lignes == null || !result.Lignes.Any())
                throw new InvalidOperationException("Aucune ligne extraite");

            return result;
        }
        catch (JsonException ex)
        {
            _logger.LogError(ex, "JSON invalide: {Reponse}", reponse);
            throw new InvalidOperationException($"Format de réponse invalide: {ex.Message}");
        }
    }

    private async Task<ExtractionResult> FallbackE4BAsync(object request)
    {
        // Fallback sur E4B si E2B échoue
        var fallbackRequest = new 
        { 
            model = MODEL_E4B,
            ((dynamic)request).prompt,
            ((dynamic)request).stream,
            ((dynamic)request).options 
        };
        
        var response = await _httpClient.PostAsJsonAsync("/api/generate", fallbackRequest);
        var result = await response.Content.ReadFromJsonAsync<OllamaResponse>();
        return ParserReponseGemma4(result!.Response, TypeDocument.BonLivraison);
    }
